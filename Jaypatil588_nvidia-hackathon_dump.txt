4. Deploying to Vercel
To deploy this project to Vercel, you need to configure it as a monorepo.

Create vercel.json:
Create a new file named vercel.json in the root of your project (at the same level as your frontend and backend folders).

Add Configuration:
Copy the provided vercel.json content into this new file. This file tells Vercel:

builds: How to build your Vite app into a static site and how to build your Flask app into a serverless function.

rewrites: To forward any request that starts with /api/ to your Flask backend. All other requests will be served your React app.

Update Frontend API Calls:
Ensure that in your frontend/src/App.jsx file, the fetch request uses a relative path:

// This path works for both local development and production on Vercel
const response = await fetch('/api/hello');

Push to Git:
Commit your code and push it to a GitHub, GitLab, or Bitbucket repository.

Deploy on Vercel:

Go to your Vercel dashboard.

Click "Add New..." -> "Project".

Import the Git repository you just pushed to.

Vercel will automatically detect the vercel.json file and build and deploy your project. No other settings should be needed.

# backend/api/index.py

from datetime import datetime, timezone
import os
from typing import Any, Dict

from flask import Flask, jsonify, send_from_directory
from flask_cors import CORS


def _health_payload() -> Dict[str, Any]:
    """Builds a standard health response for observability tests."""
    return {
        "status": "ok",
        "environment": os.getenv("DEPLOYMENT_ENV", "development"),
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "version": os.getenv("RELEASE_SHA", "local"),
    }


def create_app() -> Flask:
    # --- START OF CHANGE ---
    # Construct a robust, absolute path to the 'frontend/dist' directory.
    # This prevents pathing issues in the Vercel serverless environment.
    project_root = os.path.abspath(os.path.join(os.path.dirname(__file__), '..', '..'))
    static_folder_path = os.path.join(project_root, 'frontend', 'dist')
    # --- END OF CHANGE ---

    app = Flask(
        __name__,
        static_folder=static_folder_path, # Use the absolute path
        static_url_path='/'
    )

    allowed_origins = os.getenv("ALLOWED_ORIGINS", "*")
    CORS(app, resources={r"/api/*": {"origins": allowed_origins}})

    @app.get("/")
    def serve_frontend():
        return send_from_directory(app.static_folder, "index.html")

    @app.get("/api/health")
    def health():
        return jsonify(_health_payload())

    @app.get("/api/ping")
    def ping():
        return jsonify({"pong": True, "details": _health_payload()})

    @app.errorhandler(404)
    def not_found(e):
        return send_from_directory(app.static_folder, "index.html")

    return app


app = create_app()

import { useEffect, useState } from 'react'
import './App.css'

function App() {
  const [apiStatus, setApiStatus] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    let cancelled = false

    async function fetchHealth() {
      try {
        const response = await fetch('/api/health')
        if (!response.ok) {
          throw new Error(`Request failed with status ${response.status}`)
        }
        const data = await response.json()
        if (!cancelled) {
          setApiStatus(data)
        }
      } catch (err) {
        if (!cancelled) {
          setError(err.message || 'Unknown error')
        }
      } finally {
        if (!cancelled) {
          setLoading(false)
        }
      }
    }

    fetchHealth()

    return () => {
      cancelled = true
    }
  }, [])

  return (
    <main className="layout">
      <section className="panel">
        <h1>GitHub PR Review Agent</h1>
        <p className="intro">
          This starter view confirms the frontend build pipeline and backend API
          are reachable in Vercel deployments.
        </p>

        <div className="status-card">
          <h2>Backend Health</h2>
          {loading && <p className="status loading">Checking API status…</p>}
          {error && (
            <p className="status error">
              Unable to reach API: <span>{error}</span>
            </p>
          )}
          {!loading && !error && apiStatus && (
            <>
              <p className="status ok">API responding with status “{apiStatus.status}”.</p>
              <dl className="status-list">
                <div>
                  <dt>Environment</dt>
                  <dd>{apiStatus.environment}</dd>
                </div>
                <div>
                  <dt>Version</dt>
                  <dd>{apiStatus.version}</dd>
                </div>
                <div>
                  <dt>Timestamp</dt>
                  <dd>{apiStatus.timestamp}</dd>
                </div>
              </dl>
            </>
          )}
        </div>

        <div className="next-steps">
          <h3>Next Steps</h3>
          <ol>
            <li>Replace this placeholder UI with your dashboard components.</li>
            <li>Expand the Flask API under <code>backend/api/</code>.</li>
            <li>Update deployment environment variables in Vercel.</li>
          </ol>
        </div>
      </section>
    </main>
  )
}

export default App
